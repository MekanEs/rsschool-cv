function dirReduc(arr){

        let z = []

        for (let i=0; i<arr.length; i++){
            if(z.length===0){
                z.push(arr[i])
            } else if(arr[i]==="NORTH"&&z[z.length-1]==="SOUTH"){
                z.pop()
            }else if(arr[i]==="SOUTH"&&z[z.length-1]==="NORTH"){
                z.pop()
            }else if(arr[i]==="EAST"&&z[z.length-1]==="WEST"){
                z.pop()
            }else if(arr[i]==="WEST"&&z[z.length-1]==="EAST"){
                z.pop()
            } else{
                z.push(arr[i])
            }

        }
        return z
    }
